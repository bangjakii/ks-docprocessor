# KS doc processor — dependencies
# Inti pipeline
anthropic                 # klasifikasi (process_docs/ingest) + vision OCR + LLM field extraction
python-dotenv             # baca .env

# Pinecone (integrated embedding, multilingual-e5-large — TANPA OpenAI)
pinecone                  # sync client (extra [asyncio] tidak dipakai)

# PDF → teks, render & potong
pdfplumber                # ekstrak teks PDF digital
pdf2image                 # render PDF→gambar + hitung halaman (butuh poppler terinstall)
pypdf                     # potong/split halaman bundel (ingest split + cut search-time)
pytesseract               # OCR tulisan cetak (butuh Tesseract-OCR terinstall)
pillow                    # baca file gambar (.jpg/.png/…) utk OCR
tqdm                      # progress bar ingest (opsional, ada fallback)

# Office modern (ekstrak teks, gratis)
python-docx               # .docx
openpyxl                  # .xlsx/.xlsm

# Catatan: poppler & Tesseract adalah binary eksternal (set POPPLER_PATH / TESSERACT_PATH di .env).
# Legacy .doc/.xls hanya dapat fallback record (nama+metadata), tidak diekstrak teksnya.
